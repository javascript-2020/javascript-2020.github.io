var xhr = new XMLHttpRequest(); // readyState will be 0 console.log("UNSENT",xhr.readyState); xhr.open('GET','/api',true); // readyState will be 1 console.log("OPENED",xhr.readyState); xhr.onprogress = ()=>{ // readyState will be 3 console.log("LOADING",xhr.readyState); }//onprogress xhr.onload = ()=>{ // readyState will be 4 console.log("DONE",xhr.readyState); }//onload xhr.send(null);
